271 void printLookup() {
272  static elapsedMillis timeElapsed;
273  noInterrupts();
274  Serial.flush();
275  uint64_t ullen = 0;
276  uint8_t *bytes = (uint8_t *)&(ullen);
277  Serial.write(bytes, 8);
278  ullen = log_lookup_len;
279  Serial.write(bytes, 8);
280  ullen = log_lookup_uncmp_len;
281  Serial.write(bytes, 8);
282  timeElapsed = 0;
283  for (size_t i = 0; i < log_lookup_pad_len; i += 8) {
284  Serial.write(log_lookup + i, 8);
285  Serial.flush();
286  while (!Serial.available()) {
287  if (timeElapsed > 31000) { // Entire process should take less than 30s
288  interrupts();
289  return;
290  }
291  }
292  Serial.read();
293  }
294  interrupts();
295 }
