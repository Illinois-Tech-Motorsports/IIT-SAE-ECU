18 def bin2c(data: bytes, var_name: str) -> str:
19  """Generate a C char array from bytes
20  Args:
21  data (bytes): The bytes
22  var_name (str): The C variable name
23 
24  Returns:

27  data = data.hex(" ", 1).split(" ")

29  out: list[str] = []
