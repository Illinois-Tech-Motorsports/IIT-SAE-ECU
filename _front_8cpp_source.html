149 static void pushCanMessage() {
150  uint32_t address;
151  char buffer[8];
152  size_t c = 0;
153  c += Serial.readBytes((char *)&address, 4);
154  c += Serial.readBytes(buffer, 8);
155  if (c != 12) {
156  Log.w(ID, "Did not read correct number of Bytes, not pushing message", 12);
157  return;
158  }
159  Log.d(ID, "Pushing Message", address);
160  Canbus::sendData(address, (uint8_t *)buffer);
161 }

168 static void sendEchoMessage() {
169  uint32_t delay;

172  size_t c = 0;

178  return;

181  Echo::echo(delay, address, (uint8_t *)buffer);

266  Cmd::setCommand(COMMAND_TOGGLE_CANBUS_SNIFF, toggleCanbusSniffer);
267  Cmd::setCommand(COMMAND_SEND_ECHO, sendEchoMessage);
268  Cmd::setCommand(COMMAND_TOGGLE_REVERSE, toggleMotorDirection);

271  Heartbeat::beginReceiving();
272 #ifdef CONF_ECU_DEBUG
273  Mirror::setup();
274 #endif
275 
276  static bool hasBeat = false;
277 
278  Log.d(ID, "Delaying 2 sec");
279  Serial.flush();
280  blinkStart();
281  Pins::setInternalValue(PINS_INTERNAL_SYNC, 1);
282  blinkStart();
283  Serial.flush();
284 #if TESTING != FRONT_ECU
285  while (!Heartbeat::checkBeat()) {
286  delay(500);
287  Cmd::receiveCommand();

296  testValues();
297 #else

299  timeElapsedHigh = 0;

301 #ifndef SILENT
302  Log(ID, "Current Motor Speed:", MC::motorSpeed(), true);

311 #endif
312  }

314  timeElapsedMidHigh = 0;
315  updateCurrentState();

317  }
318  if (timeElapsedMidLow >= INTERVAL_MED_LOW_PRIORITY) { // MedLow priority updates
319  timeElapsedMidLow = 0;
