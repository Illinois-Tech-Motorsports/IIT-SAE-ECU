141 async def map_lines(self, function: Callable[[str], str]) -> None:
142 """Map a function to each line of this file
143
144 Args:
145 function (Callable[[str], None]): Function that takes in a str and returns a str
146 """
147 temp_path = self.working_pathworking_path + ".__Lock"
148 line_no = 1
149 newline = ""
150
151 with open(self.pathpath, "r", encoding="utf-8") as original, open(temp_path, "w", encoding="utf-8") as new_file:
152 for line in original:
153 try:
154 newline = await function(line)
155 new_file.buffer.write(newline.encode("utf-8"))

160 line_no += 1
161 self.modifiedmodified = not Util.syncFile(temp_path, self.offsetoffset, self.rawpathrawpath, self.working_pathworking_path)

164 async def match_log_mapping(self, line: str) -> str:
165 """Maps a source file line to Log related syntax
166
167 Args:
168 line (str): The line to scan
169
170 Raises:
171 ScriptException: On invalid Log related syntax
172
173 Returns:
174 str: Reformatted line
175 """
176 newline: str = line
177
178 if IGNORE_KEYWORD in newline: # Return if this line has the ignore keyword
179 return newline
180
181 scan_sequence = [
182 (REGEX.CALL_VS, self.line_vsx),
183 (REGEX.CALL_VSV, self.line_vsx),
184 (REGEX.CALL_SS, self.line_ssx),
185 (REGEX.CALL_SSV, self.line_ssx),
186 (REGEX.SPECIAL_PASS, self.line_special),
187 (REGEX.TAG_PASS, self.line_tag),
188 ]
189
190 for reg, func in scan_sequence:
191 matches = re.findall(reg, line)
192 if len(matches) != 0:
193 return await func(line, matches[0])
194
195 fail_sequence = [
196 (REGEX.TAG_FAIL, Error.MalformedTAGDefinitionException),
197 (REGEX.CALL_ERR_LITERAL, Error.MalformedLogCallException),
198 (REGEX.CALL_ERR_BLANK, Error.BlankTAGException),
199 ]
200
201 for reg, exception in fail_sequence:
202 err = re.findall(reg, line)
203 if len(err) != 0:
204 err = err[0]
205 raise exception(err[0] + Text.error(err[1]) + err[2])
206
207 return newline
