39class Peeker:
40
41 generator: Generator = None
42 peek: TokenInfo = None
43 empty = False
44
45 def __init__(self, generator):
46 self.generator = generator
47 try:
48 self.peek = next(self.generator)
49 except StopIteration:
50 self.emptyempty = True
51
52 def __iter__(self):
53 return self
54
55 def __next__(self):
56 """
57 Return the self.peek element, or raise StopIteration
58 if empty
59 """
60 if self.emptyempty:
61 raise StopIteration()
62 to_return = self.peek
63 try:
64 self.peek = next(self.generator)
65 except StopIteration:
66 self.peek = None
67 self.emptyempty = True
68 return to_return
69
70 def tkn(self) -> str:
71 """Returns the next formatted token
72
73 Returns:
74 str: next token
75 """
76 return norm(next(self))
77
78 def num(self) -> float:
79 """Gets the next number, uses up to two tokens
80
81 Returns:
82 float: number as a float, error if not found
83 """
84 frst = next(self).string
85
86 if frst == "-" or frst == "+":
87 frst = frst + next(self).string
88
89 return float(frst)
90
91 def desc(self) -> str | None:
92 """Gets the description, provided it exists, uses up as many tokens that remain
93
94 Returns:
95 str | None: The description as a string, or None if it does not exist
96 """
97 nxt = self.peek
98 if nxt and nxt.string == ":":
99 next(self)
100 return " ".join([t.string.strip('"') for t in self][:-2])
101 return None
102
103 def node(self) -> str | None:
104 """Gets the name of a node, provided it exists, uses one token
105
106 Returns:
107 str | None: The node as a string, or None if it does not exist
108 """
109 nxt = self.peek
110 if nxt.string == "|":
111 return None
112 return norm(next(self))

186 self.f_type = f_type
187 self.offsetoffset = (v_max == self.v_maxv_max) and (v_min == self.v_minv_min)
188 self.calccalc = (scale == self.scalescale) and (offset == self.offsetoffset)

254class Message(Entry):
255 """Message class for defining and verifying messages"""
256
257 c_node: Entry = None
258 can_id: int = 0
259 signals: List[Signal] = None
260
261 def __init__(self, uid: int, line: int, name: str, description: str, c_node: Entry, can_id: int) -> None:
262 super().__init__(uid, line, name, description)
263 self.c_node = c_node
264 self.can_id = int(can_id, 0)
265 self.signals = []
266
267 def __repr__(self) -> str:
268 desc = f" : {self.description}" if self.description else ''
269 fnl = f"MSG {self.can_id} {self.name}{desc}\n"
270 for sig in self.signals:
271 fnl += repr(sig)
272 return fnl + '\n'
273
274 def __str__(self) -> str:
275 sigs = listify([str(x) for x in self.signals], "\n ")
276 blk = "|".join([str(x.form.f_type.bits) for x in self.signals])
277 return f"{self.can_id} {self.name} :\n {sigs}\n |{blk}|"
278
279 def finalize(self, formats: Dict[str, Format], nodes: Dict[str, Entry], g_pins: Set[int]) -> None:
280 """Verifies this Message, matching it's sender, should only be called once
281
282 Args:
283 formats (Dict[str, Format]): Defined Formats
284 nodes (List[str]): List of all nodes
285 """
286
287 shift = 64
288 for sig in self.signals:
289 sig.finalize(self.c_node, formats)
290 shift -= sig.form.f_type.bits
291 sig.set_pos(shift)
292
293 assert sum([sig.form.f_type.bits for sig in self.signals]) <= 64, f"Message exceeds 64 bits : {self.line}| {self.name}"
294
295 self.c_node.add_address(self.can_id)
296
297 for node in nodes.values():
298 if node is not self.c_node:
299 assert self.can_id not in node.addr_space, f"Can id already in use in another address space {self.can_id} : {node.name}"

302class Node(Entry):
303 """Node class for defining nodes"""
304
305 sources: dict[str, Entry] = None
306 names: set[str] = None
307 addr_space: set[int] = None
308
309 def __init__(self, uid: int, line: int, name: str, description: str) -> None:
310 super().__init__(uid, line, name, description)
311 self.names = set()
312 self.sources = {}
313 self.addr_space = set()
314
315 def add_address(self, addrs: Sequence | int) -> None:
316 if isinstance(addrs, Sequence):
317 for addr in addrs:
318 self.addr_space.add(addr)
319 else:
320 self.addr_space.add(addrs)
321
322 def __repr__(self) -> str:
323 desc = f" : {self.description}" if self.description else ''
324 fnl = f"NODE {self.name}{desc}\n"
325 for sig in self.sources:
326 fnl += repr(sig)
327 return fnl + '\n'
328
329 def __getitem__(self, item: str) -> Entry:
330 assert isinstance(item, str), "Can only index with strings"
331 return self.sources[item]
332
333 def __str__(self) -> str:
334 srcs = listify([str(x) for x in self.sources], "\n ")
335 return f"{self.name} :\n {srcs}\n"
336
337 def insert(self, __e: Entry) -> None:
338 assert __e.name not in self.names, f"Source has already been defined in namespace {self.name} : {__e.name}"
339 self.names.add(__e.name)
340 self.sources[__e.name] = __e
341
342 def finalize(self, formats: Dict[str, Format], nodes: Dict[str, Entry]) -> None:
343 g_pins: Set[int] = set()
344 srcs = [x for x in self.sources.values()]
345 for src in srcs:
346 if not isinstance(src, Signal):
347 src.finalize(formats, nodes, g_pins)
348 for k, v in [x for x in self.sources.items()]:
349 if isinstance(v, Signal) or isinstance(v, SYNC):
350 del self.sources[k]
351 for n in nodes.values():
352 if n is not self:
353 assert self.addr_space.isdisjoint(n.addr_space), "Overlapping address spaces"
354
355 def mark(self):
356 """Explicilty mark synced items that this node does not own"""
357 for k, v in [x for x in self.sources.items()]:
358 if v.c_node is not self:
359 self.sources['_'+v.c_node.name+'_'+k] = v
360 del self.sources[k]

463def parse(file: TextIO) -> SDBC:
464 """Parses a SDBC file given an open file
465
466 Args:
467 file (TextIO): The open file to parse
468 """
469 file_name: str = None
470 version: str = None
471
472 types: Dict[str, Type] = {}
473 formats: Dict[str, Format] = {}
474 messages: Dict[int, Message] = {}
475 signals: Dict[str, Signal] = {}
476 nodes: Dict[str, Node] = {}
477
478 last_node: Node = None
479 last_msg: Message = None
480 line_no = 0
481
482 uid = 0
483
484 try:
485 for line in file.readlines():
486 line_no += 1
487 uid += 1
488 tks = Peeker(tokenize(BytesIO(line.strip(" \n").encode("utf-8")).readline))
489 next(tks)
490 sel = next(tks)
491 if sel.type == tkn.NAME:
492 match sel.string:
493 case "NAME":
494 last_node = None
495 last_msg = None
496 file_name = tks.tkn()
497 case "VERSION":
498 last_node = None
499 last_msg = None
500 version = tks.tkn() + tks.tkn()
501 case "NODE":
502 last_msg = None
503 name = tks.tkn()
504 desc = tks.desc()
505 node = Node(uid, line_no, name, desc)
506 last_node = node
507 nodes[name] = node
508 case "ADDR":
509 last_msg = None
510 lower = int(tks.tkn(), 0)
511 assert tks.tkn() == "-", "Invalid signal syntax"
512 upper = int(tks.tkn(), 0)
513 assert lower <= upper, "Invalid address range"
514 assert last_node, "No encapuslating node"
515 last_node.add_address(range(lower, upper+1))
516 case "GPIO":
517 last_msg = None
518 name = tks.tkn()
519 assert next(tks).string == ":", "Invalid signal syntax"
520 pin = tks.tkn()
521 p_type = tks.tkn()
522 io_type = tks.tkn()
523 desc = tks.desc()
524 gpio = GPIO(uid, line_no, name, desc, last_node, pin, p_type, io_type)
525 assert last_node, "No encapuslating node"
526 last_node.insert(gpio)
527 case "VIRT":
528 last_msg = None
529 name = tks.tkn()
530 assert next(tks).string == ":", "Invalid signal syntax"
531 p_type = tks.tkn()
532 io_type = tks.tkn()
533 desc = tks.desc()
534 virt = VIRT(uid, line_no, name, desc, last_node, p_type, io_type)
535 assert last_node, "No encapuslating node"
536 last_node.insert(virt)
537 case "SYNC":
538 last_msg = None
539 node = tks.tkn()
540 name = tks.tkn()
541 desc = tks.desc()
542 sync = SYNC(uid, line_no, '__SYNC:'+node+':'+name, desc, last_node, node, name)
543 assert last_node, "No encapuslating node"
544 last_node.insert(sync)
545 case "MSG":
546 can_id = tks.tkn()
547 name = tks.tkn()
548 desc = tks.desc()
549 msg = Message(uid, line_no, name, desc, last_node, can_id)
550 assert msg.can_id not in messages, f"Message CAN ID already in use : {msg.can_id}"
551 messages[msg.can_id] = msg
552 last_msg = msg
553 assert last_node, "No encapuslating node"
554 last_node.insert(msg)
555 case "SIG":
556 name = tks.tkn()
557 assert next(tks).string == ":", "Invalid signal syntax"
558 form = tks.tkn()
559 desc = tks.desc()
560 link = None
561 if tks.peek and tks.peek.string == '<':
562 assert tks.tkn() == '<', "Invalid signal syntax"
563 assert tks.tkn() == '-', "Invalid signal syntax"
564 link = tks.tkn()
565 sig = Signal(uid, line_no, name, desc, form, link)
566 assert last_msg, "Orphaned signal definition"
567 assert sig.name not in signals, f"Signal name already in use : {last_msg.name}"
568 last_msg.signals.append(sig)
569 signals[sig.name] = sig
570 assert last_node, "No encapuslating node"
571 last_node.insert(sig)
572 case "TYPE":
573 last_node = None
574 last_msg = None
575 name = tks.tkn()
576 bits = tks.tkn()
577 n_type = tks.tkn()
578 desc = tks.desc()
579 typ = Type(uid, line_no, name, desc, bits, n_type)
580 assert typ.name not in types, f"Type name already in use : {typ.name}"
581 typ.finalize()
582 types[typ.name] = typ
583 case "FORMAT":
584 last_node = None
585 last_msg = None
586 name = tks.tkn()
587 signed = next(tks).string == "-"
588 f_type = tks.tkn()
589 assert next(tks).string == "(", "Invalid signal syntax"
590 scale = tks.num()
591 assert next(tks).string == ",", "Invalid signal syntax"
592 offset = tks.num()
593 assert next(tks).string == ")", "Invalid signal syntax"
594 assert next(tks).string == "[", "Invalid signal syntax"
595 v_min = tks.num()
596 assert next(tks).string == ",", "Invalid signal syntax"
597 v_max = tks.num()
598 assert next(tks).string == "]", "Invalid signal syntax"
599 desc = tks.desc()
600 frmt = Format(uid, line_no, name, desc, signed, f_type, scale, offset, v_min, v_max)
601 assert frmt.name not in formats, f"Format name already in use : {line_no}| {frmt.name}"
602 formats[frmt.name] = frmt
603
604 for form in formats.values():
605 form.finalize(types)
606
607 for node in nodes.values():
608 node.finalize(formats, nodes)
609 node.mark()
610
611 assert len(messages) <= 1048575, "Exceeding maximum number of messages"
612
613 except Exception as err:
614 print(err, f'| line {line_no}')
615 return
616
617 print(f"{file_name} | SDBC : {version}")
618
619 return SDBC(file_name, version, nodes, types, formats, messages, signals)
