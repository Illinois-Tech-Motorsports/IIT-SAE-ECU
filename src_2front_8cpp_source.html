22static void pushCanMessage() {
23 uint32_t address;
24 char buffer[8];
25 size_t c = 0;
26 c += Serial.readBytes((char *)&address, 4);
27 c += Serial.readBytes(buffer, 8);
28 if (c != 12) {
29 Log.w(ID, "Did not read correct number of Bytes, not pushing message", 12);
30 return;
31 }
32 Log.d(ID, "Pushing Message", address);
33 Canbus::sendData(address, (uint8_t *)buffer);
34}
