94 bool ECUStates::PreCharge_State::voltageCheck() {
95  int16_t BMSVolt = BMS_DATA_Buffer.getShort(2) / 10; // Byte 2-3: Pack Instant Voltage
96  int16_t MC0Volt = MC0_VOLT_Buffer.getShort(0) / 10; // Bytes 0-1: DC BUS MC Voltage
97  int16_t MC1Volt = MC1_VOLT_Buffer.getShort(0) / 10; // Bytes 0-1: DC BUS MC Voltage
98 
99  return 0.62 * BMSVolt <= (MC0Volt + MC1Volt) / 2;
100 }

108 State::State_t *ECUStates::PreCharge_State::run(void) { // NOTE: Low = Closed, High = Open
109  Log.i(ID, "Loading Buffers");
110  getBuffers();
111  Log.i(ID, "Precharge running");
112 
113  if (FaultCheck()) {
114  Log.e(ID, "Precharge Faulted before precharge");
115  return &ECUStates::FaultState;
116  }
117 
118  // NOTE: Assuming Air2 is correct
119  Log.w(ID, "Closing Air2 and Precharge Relay and opening Air1");
120  Pins::setPinValue(PINS_BACK_AIR2, PINS_ANALOG_HIGH);
121  Pins::setPinValue(PINS_BACK_PRECHARGE_RELAY, PINS_ANALOG_HIGH);
122  Pins::setPinValue(PINS_BACK_AIR1, LOW);
123 
124  if (FaultCheck()) {
125  return PreCharFault();
126  }
127 
128  elapsedMillis timeElapsed;
129 
130  Log.d(ID, "Running precharge loop");
131 
132  while (timeElapsed <= 10000) {
133  if (timeElapsed >= 1000 && voltageCheck()) { // NOTE: will always pass if submodules are disconnected from CAN net
134  Log.w(ID, "Opening precharge relay");
135  Pins::setPinValue(PINS_BACK_PRECHARGE_RELAY, LOW);
136  Log.i(ID, "Precharge Finished, closing Air1");
137  Pins::setPinValue(PINS_BACK_AIR1, PINS_ANALOG_HIGH);
138  return &ECUStates::Idle_State;
139  }
140  }
141 
142  Log.e(ID, "Precharge timed out");
143  return PreCharFault();

238 State::State_t *ECUStates::Driving_Mode_State::run(void) {
239  Log.i(ID, "Driving mode on");
240  Log.i(ID, "Cooling on");
241  carCooling(true);
242 
243  elapsedMillis controlDelay;
244  size_t counter = 0;
245 
246  Log.i(ID, "Loading Buffers");
247  MC0_VOLT_Buffer.init();
248  MC1_VOLT_Buffer.init();
249 
250  Log.i(ID, "Stopping MC heartbeat");
251  MC::enableMotorBeating(false);
252 
253  Log.d(ID, "Sending Fault reset to MCs complete");
254  MC::clearFaults(); // Clear fault if any
255 
256  Log.d(ID, "Entering drive loop");
257  while (true) {
258  if (controlDelay > 5) { // NOTE: Each data frame is 89 bits long thus at 250kbps the MC buses can handle a maximum of 2808 messages per second
259  controlDelay = 0;
260 
261  static bool runReverse = false;
262  if (counter % 5 == 0 && Pins::getCanPinValue(PINS_INTERNAL_REVERSE) != runReverse) {
263  runReverse = Pins::getCanPinValue(PINS_INTERNAL_REVERSE);
264  MC::setDirection(!runReverse);
265  }
266 
267  if (Fault::softFault() || Fault::hardFault()) { // FIXME: are motor controller faults actually being picked up?
268 #if TESTING != BACK_ECU
269  return DrivingModeFault();
270 #endif
271  }
272 
273  if (((MC0_VOLT_Buffer.getShort(0) / 10) + (MC1_VOLT_Buffer.getShort(0) / 10)) / 2 < 90) { // 'HVD Fault'
274 #if TESTING != BACK_ECU
275  Log.e(ID, "'HVD Fault' MC voltage < 90");
276  return DrivingModeFault();
277 #endif
278  }
279 
280  int breakVal = Pins::getCanPinValue(PINS_FRONT_BRAKE);
281  int steerVal = Pins::getCanPinValue(PINS_FRONT_STEER);
282 
283  Pins::setPinValue(PINS_BACK_BRAKE_LIGHT, PINS_ANALOG_HIGH * ((float)breakVal / PINS_ANALOG_MAX > 0.04f));
284 
285  int pedal0 = Pins::getCanPinValue(PINS_FRONT_PEDAL0);
286  int pedal1 = Pins::getCanPinValue(PINS_FRONT_PEDAL1);
287 
288  int pAVG = (pedal0 + pedal1) / 2;
289 
290  // NOTE: pedal has a threshold value
291  if (pAVG >= 100 && (float)abs(pedal1 - pedal0) / PINS_ANALOG_HIGH > 0.1f) {
292  Log.e(ID, "Pedal value offset > 10%");
293 #if TESTING != BACK_ECU
294  return DrivingModeFault();
295 #endif
296  }
297 
298  MC::setTorque(pAVG, breakVal, steerVal);
299 #if TESTING != BACK_ECU
300  if (++counter > 128) {
301 #else
302  if (++counter > 10) {
303 #endif
304  counter = 0;
305  Log.i(ID, "Aero servo position:", Aero::getServoValue(), true);
306  if (Fault::softFault()) {
307  Fault::logFault();
308  }
309  }
310 
311  Log.i(ID, "Last MC0 Torque Value:", MC::getLastTorqueValue(true), true);
312  Log.i(ID, "Last MC1 Torque Value:", MC::getLastTorqueValue(false), true);
313 
314  Aero::run(breakVal, steerVal);
315  }
316 
317  if (!Pins::getCanPinValue(PINS_FRONT_BUTTON_INPUT_OFF)) {
318  Log.w(ID, "Going back to Idle state");
319  break;
320  }
321  }
322 
323  Log.i(ID, "Starting MC heartbeat");
324  MC::enableMotorBeating(true);
325 
326  carCooling(false);
327 
328  Log.i(ID, "Driving mode off");
329  return &ECUStates::Idle_State;
330 }
