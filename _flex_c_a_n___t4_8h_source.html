37 typedef struct CAN_message_t {
38  uint32_t id = 0; // can identifier
39  uint16_t timestamp = 0; // FlexCAN time when message arrived
40  uint8_t idhit = 0; // filter that id came from
41  struct {
42  bool extended = 0; // identifier is extended (29-bit)
43  bool remote = 0; // remote transmission request packet type
44  bool overrun = 0; // message overrun
45  bool reserved = 0;
46  } flags;
47  uint8_t len = 8; // length of data
48  uint8_t buf[8] = {0}; // data
49  int8_t mb = 0; // used to identify mailbox reception
50  uint8_t bus = 0; // used to identify where the message came from when events() is used.
51  bool seq = 0; // sequential frames
52 } CAN_message_t;
53 
54 typedef struct CANFD_message_t {
55  uint32_t id = 0; // can identifier
56  uint16_t timestamp = 0; // FlexCAN time when message arrived
57  uint8_t idhit = 0; // filter that id came from
58  bool brs = 1; // baud rate switching for data
59  bool esi = 0; // error status indicator
60  bool edl = 1; // extended data length (for RX, 0 == CAN2.0, 1 == FD)
61  struct {
62  bool extended = 0; // identifier is extended (29-bit)
63  bool overrun = 0; // message overrun
64  bool reserved = 0;
65  } flags;
66  uint8_t len = 8; // length of data
67  uint8_t buf[64] = {0}; // data
68  int8_t mb = 0; // used to identify mailbox reception
69  uint8_t bus = 0; // used to identify where the message came from when events() is used.
70  bool seq = 0; // sequential frames
71 } CANFD_message_t;
