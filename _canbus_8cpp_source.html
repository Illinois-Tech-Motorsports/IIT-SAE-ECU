100static void _receiveCan(const CAN_message_t &msg) { // FIXME: potential issue where checking semaphore freezes, more testing needed
101 if (addressSemaphore == msg.id) { // Throw data away, we are already processing previous msg
102#ifdef CONF_ECU_DEBUG
103 Log.w(ID, "Discarding can msg", msg.id);
104#endif
105 return;
106 }
107
108 CanAddress_t *addr = _getAddress(msg.id);
109
110 if (addr == invalidAddress) {
111 return;
112 }
113
114 volatile uint8_t *arr = addr->buffer;
115 arr[0] = msg.buf[0];
116 arr[1] = msg.buf[1];
117 arr[2] = msg.buf[2];
118 arr[3] = msg.buf[3];
119 arr[4] = msg.buf[4];
120 arr[5] = msg.buf[5];
121 arr[6] = msg.buf[6];
122 arr[7] = msg.buf[7];
123 if (addr->callback)
124 addr->callback(msg.id, addr->buffer);
125}

169void getData(const uint32_t address, uint8_t buf[8]) {
170 CanAddress_t *addr = _getAddress(address);
171 if (addr != invalidAddress && !addr->flow) { // 0 == incoming
172 setSemaphore(address); // Semaphore needed to ensure interrupt does not replace data mid transfer
173 copyVolatileCanMsg(addr->buffer, buf);
174 clearSemaphore();
175 return;
176 }
177 Log.e(ID, "Given address is not incoming or is invalid:", address);
178}

201void pushData(const uint32_t address) {
202#ifdef CONF_ECU_DEBUG
203 CanAddress_t *addr = _getAddress(address);
204 if (addr == invalidAddress) {
205 Log.e(ID, "Address has not been allocated: ", address);
206 return;
207 }
208 copyVolatileCanMsg(addr->buffer, send.buf);
209#else
210 copyVolatileCanMsg(_getAddress(address)->buffer, send.buf);
211#endif
212 send.id = address;
213 _pushSendMsg();
214}
