1"""
2@file Pre_Build.py
3@author IR
4@brief This script preprocesses source files for use with Log_t
5@version 0.1
6@date 2020-11-11
7
8@copyright Copyright (c) 2022
9
10This script works by first duplicating source files to the build folder. \n
11Then it scans each file for calls to a Log function and modifies them as follows. \n
12
13If the call has a string for `LOG_TAG` parameter, give that string a unique integer ID and replace it with that integer. \n
14If the ID is not a string, leave the variable alone. \n
15
16Replace the call's `LOG_MSG` string with a unique ID as well. \n
17NOTE: The `LOG_MSG` parameter must always be an inline string. \n
18
19`LOG_TAG`s and `LOG_MSG`s do not share IDs. \n
20
21Eg.
22```
23Log(ID, "My Message"); -> Log(ID, 1);
24Log("My Str ID", "My Message"); -> Log(1, 1);
25```
26
27Calls to Log functions also have the option to send a number with a third parameter. \n
28
29Eg.
30```
31Log("My Str ID", "My Message", 56); -> Log(1, 1, 56);
32Log(ID, "My Message", A_Num_Var); -> Log(ID, 1, A_Num_Var);
33```
34
35Declarations of `LOG_TAG` also have their strings replaced with a unique ID. \n
36NOTE: Definition of `LOG_TAG`s must always be an inline string. \n
37
38Eg.
39```
40LOG_TAG TAG = "Logging Tag"; -> LOG_TAG TAG = 2;
41```
42
43A special case has been made to also allocate and replace string that call the following macro
44
45```
46_LogPrebuildString(x)
47```
48
49Where x is the string, it will be given a unique ID and replaced with said ID as if it were being called by a Logging function.
50This is useful where one wishes to generate log functions using the C preprocessor.
51Note, however, if this script is not run the macro should still allow everything to compile normally, leaving the string untouched
52
53"""

110def getLibraryBlacklist() -> dict[str, list]:
111 """Get the library folder blacklist based on core model
112
113 Returns:
114 dict[str, list]: folder blacklist dict
115 """
116 blacklist: dict[str, list] = {}
117 with open(BLACKLIST, "r", encoding="utf-8") as file:
118 currentModel = ""
119 for line in file.readlines():
120 if line[0] == ".":
121 currentModel = line.split(" ")[0][1:]
122 if currentModel not in blacklist:
123 blacklist[currentModel] = []
124 else:
125 for token in line.split(" "):
126 token = token.strip(" \n")
127 if not token or token[0] == "#":
128 break
129 elif os.path.exists(join_path(os.path.dirname(BLACKLIST), token)):
130 blacklist[currentModel].append(join_path(os.path.dirname(BLACKLIST), token))
131 return blacklist

144def hashFile(filePath):
145 if os.path.exists(filePath):
146 if available_ram() <= LOW_RAM:
147 sha256 = hashlib.sha256()
148 with open(filePath, "rb") as f:
149 while True:
150 data = f.read(BUF_SIZE)
151 if not data:
152 break
153 sha256.update(data)
154 return sha256.digest()
155 else:
156 with open(filePath, "rb") as f:
157 return hashlib.sha256(f.read()).hexdigest()
158 return ""

388 async def walkLines(self, function):
389 temp_path = self.workingPath + ".__Lock"
390 line_no = 1
391 newline = ""
392 with open(self.path, "r", encoding="utf-8") as f1, open(temp_path, "w", encoding="utf-8") as f2:
393 if self.rawpath.startswith(LIB_PATH) and self.name != LIB_FILE: # Ignore log library source files
394 f2.writelines(f1.readlines())
395 else:
396 for line in f1:
397 try:
398 newline = await function(line)
399 f2.buffer.write(newline.encode("utf-8"))
400 except Exception as e: # If prev exception was about IO then oh well
401 self.newError(e, self.path, line_no)
402 f2.buffer.write(line.encode("utf-8"))
403 finally:
404 line_no += 1
405 self.modified = not syncFile(temp_path, self.offset, self.rawpath, self.workingPath)
406 os.remove(temp_path)

408 async def findLogMatch(self, line: str):
409 newline: str = line
410
411 if IGNORE_KEYWORD in newline: # Return if this line has the ignore keyword
412 return newline
413
414 SPECIAL = re.findall(FIND_SPECIAL_REGEX, line)
415 if SPECIAL:
416 newline = await self.SPECIAL_STR(line, SPECIAL[0])
417 else:
418 VS = re.findall(FIND_CALL_REGEX_VS, line)
419 if len(VS) != 0: # ¯\_(ツ)_/¯
420 newline = await self.VSX(line, VS[0])
421 else:
422 TAG_GOOD = re.findall(FIND_TAG_DEF_REGEX_GOOD, line)
423 if TAG_GOOD:
424 newline = await self.NEW_TAG(line, TAG_GOOD[0])
425 else:
426 VSV = re.findall(FIND_CALL_REGEX_VSV, line)
427 if len(VSV) != 0:
428 newline = await self.VSX(line, VSV[0])
429 else:
430 TAG_BAD = re.findall(FIND_TAG_DEF_REGEX_BAD, line)
431 if TAG_BAD:
432 TAG_BAD = TAG_BAD[0]
433 raise MalformedTAGDefinitionException(TAG_BAD[0] + Text.error(TAG_BAD[1]) + TAG_BAD[2])
434 else:
435 BAD = re.findall(FIND_CALL_REGEX_BAD, line)
436 if BAD:
437 BAD = BAD[0]
438 raise MalformedLogCallException(BAD[0] + Text.error(BAD[1]) + BAD[2])
439 else:
440 SS = re.findall(FIND_CALL_REGEX_SS, line)
441 if len(SS) != 0:
442 newline = await self.SSX(line, SS[0])
443 else:
444 SSV = re.findall(FIND_CALL_REGEX_SSV, line)
445 if len(SSV) != 0:
446 newline = await self.SSX(line, SSV[0])
447 return newline

563 class TextIO(io.TextIOWrapper):
564 def __init__(
565 self,
566 func,
567 buffer: IO[bytes],
568 encoding: str = ...,
569 errors: Optional[str] = ...,
570 newline: Optional[str] = ...,
571 line_buffering: bool = ...,
572 write_through: bool = ...,
573 ):
574 super(ThreadedProgressBar.TextIO, self).__init__(buffer, encoding, errors, newline, line_buffering, write_through)
575 self.func = func
576
577 def write(self, s):
578 self.func(s.strip("\n "))

580 def getOriginal(self):
581 return super()
582
583 def __init__(self, maxcount, prefix):
584 self.maxcount = maxcount
585 self.stdout = sys.stdout
586 self.rename(prefix)

588 def rename(self, prefix):
589 mx_sz = len(self.formatStr.format(prefix, " " * self.bar_len, 100.0, "%"))
590 self.bar_len = min(int(os.get_terminal_size().columns - 1 - (mx_sz / 1.25)), mx_sz)
591 self.bar_len = self.bar_len if self.bar_len > 0 else 0
592 self.prefix = prefix
593
594 def reset(self, maxcount, prefix):
595 self.maxcount = maxcount
596 self.rename(prefix)
597 self.counter = 0
598
599 def _newLine(self, String):
600 self.Lines.add(String)
601
602 def _progress(self, count, total, prefix="", printString: str = ""):
603 if total > 0:
604 filled_len = int(round(self.bar_len * count / float(total)))
605
606 percents = round(100.0 * count / float(total), 1)
607 bar = "█" * filled_len + "░" * (self.bar_len - filled_len)
608
609 proStr = self.formatStr.format(prefix, bar, percents, "%")
610 if len(printString) > 0:
611 self.stdout.write(" " * (os.get_terminal_size().columns - 1))
612 self.stdout.write("\033[F")
613 printString = printString.strip(" \n")
614 spacer = " " * (os.get_terminal_size().columns - 1 - len(printString))
615 self.stdout.write(f"{printString}{spacer}"[: os.get_terminal_size().columns - 1])
616 self.stdout.write("\n")
617 self.stdout.write(proStr)
618 self.stdout.flush()
619
620 def _printThread(self):
621 while self.run or len(self.Lines) > 0:
622 if len(self.Lines) > 0:
623 self._progress(self.counter, self.maxcount, self.prefix, self.Lines.pop())
624 else:
625 self._progress(self.counter, self.maxcount, self.prefix)

627 def start(self):
628 self.wrapper = ThreadedProgressBar.TextIO(
629 self._newLine,
630 sys.stdout.buffer,
631 sys.stdout.encoding,
632 sys.stdout.errors,
633 sys.stdout.newlines,
634 sys.stdout.line_buffering,
635 sys.stdout.write_through,
636 )
637 sys.stdout = self.wrapper
638 self.printer = threading.Thread(target=self._printThread)
639 self.printer.start()
640
641 def progress(self):
642 self.counter += 1
643
644 def finish(self):
645 self.run = False
646 self.printer.join()
647 self._progress(self.counter, self.maxcount, self.prefix)
648 self.wrapper.flush()
649 sys.stdout = self.wrapper.getOriginal()
650 print()
