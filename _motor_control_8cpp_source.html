63int32_t motorSpeed(int motor) {

66 if (!MC0_lock.locked || !MC1_lock.locked) {
67 Log.w(ID, "Unable to lock buffers for speed", lastspeed);
68 return lastspeed;
69 }
70 int16_t MC_Rpm_Val_0 = -MC0_RPM_Buffer->getShort(2); // Bytes 2-3 : Angular Velocity // NOTE: motor is negative?
71 int16_t MC_Rpm_Val_1 = MC1_RPM_Buffer->getShort(2); // Bytes 2-3 : Angular Velocity
72 float MC_Spd_Val_0 = (float)(((MC_Rpm_Val_0 / CONF_CAR_GEAR_RATIO) * c) * 60) / 63360;
73 float MC_Spd_Val_1 = (float)(((MC_Rpm_Val_1 / CONF_CAR_GEAR_RATIO) * c) * 60) / 63360;
74
75 switch (motor) {
76 case 0:
77 return lastspeed = MC_Spd_Val_0;
78 case 1:
79 return lastspeed = MC_Spd_Val_1;
80 default:

147int sendSpeed(uint32_t MC_ADD, int speed, bool direction, bool enableBit) {
148 if (beating) {
149 Log.w(ID, "Unable to set torque, heartbeat is on");
150 return 0;
151 }
152 int setSpeed = 0, torqueLimit = 50;
153 if (speed != 0) { // max analog should be what the max pedal readout is
154 setSpeed = cMap(speed, 200, PINS_ANALOG_MAX, 0, 10); // separate func for negative vals (regen)
155 }
156
157 uint8_t *bytes = (uint8_t *)&setSpeed;
158 uint8_t *limitBytes = (uint8_t *)&torqueLimit;
159 Canbus::sendData(MC_ADD, 0, 0, bytes[0], bytes[1], direction, 0b101 * enableBit, limitBytes[0], limitBytes[1]);
160 return setSpeed;
161}
