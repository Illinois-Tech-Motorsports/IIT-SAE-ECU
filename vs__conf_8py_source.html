49 def comment_remover(text: str) -> str:
50  """Remove C style comments from a str
51 
52  Args:
53  text (str): str to remove comments from
54 
55  Returns:
56  str: text with removed comments
57  """
58 
59  def replacer(match):
60  string = match.group(0)
61  if string.startswith("/"):
62  return ""
63  return string.strip()
64 
65  pattern = r'//.*?$|/\*.*?\*/|\'(?:\\.|[^\\\'])*\'|"(?:\\.|[^\\"])*"'
66  regex = re.compile(pattern, re.DOTALL | re.MULTILINE)
67  regex = re.sub(r"\n\s*\n", "\n", re.sub(regex, replacer, text), re.MULTILINE)
68  return regex.replace("},\n}", "}\n}")
