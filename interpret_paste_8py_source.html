63def list_pastes() -> tuple:
64 """List all the pastes on the paste.ee API
65
66 Returns:
67 tuple: (Return Status, set of IDs)
68 """
69 id_list = set()
70 status = True
71 req = requests.get("https://api.paste.ee/v1/pastes", headers=HEADERS)
72 if int(req.status_code / 100) == 2:
73 print(req.status_code, req.reason)
74 else:
75 print(req.status_code, req.reason, req.text)
76 return (False, id_list)

78 values = json.loads(req.text)

169 epoch_smol = -1

173 msg = line.split(" ")

186 except KeyError as err:

190 value.sort(key=lambda x: x[0])

192 lst[0] -= epoch_smol
193 lst[0] /= 1000000000
