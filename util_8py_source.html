54def getLibraryBlacklist() -> dict[str, list]:
55 """Get the library folder blacklist based on core model
56
57 Returns:
58 dict[str, list]: folder blacklist dict
59 """
60 blacklist: dict[str, list] = {}
61 with open(LIB_BLACKLIST, "r", encoding="utf-8") as file:
62 currentModel = ""
63 for line in file.readlines():
64 if line[0] == ".":
65 currentModel = line.split(" ")[0][1:]
66 if currentModel not in blacklist:
67 blacklist[currentModel] = []
68 else:
69 for token in line.split(" "):
70 token = token.strip(" \n")
71 if not token or token[0] == "#":
72 break
73 elif os.path.exists(join_path(os.path.dirname(LIB_BLACKLIST), token)):
74 blacklist[currentModel].append(join_path(os.path.dirname(LIB_BLACKLIST), token))
75 return blacklist

81def syncFile(filePath, offset, rawpath, workingFilePath=None, suppress=False):
82 workingFilePath = workingFilePath or f"{offset}{filePath}"
83 global FILES_CHANGED
84
85 new = hashFile(filePath)
86 old = hashFile(workingFilePath)
87 if old == "":
88 old = 0
89 FILES_CHANGED = FILES_CHANGED and (new == old)
90
91 if not os.path.exists(workingFilePath) or new != old:
92 touch(f"{offset}{rawpath}")
93 shutil.copyfile(filePath, workingFilePath)
94 if not suppress:
95 print(f"Sync File: {os.path.basename(workingFilePath)}")

111def available_ram() -> bool:
112 """Get The amount of RAM available in GBs
113
114 Returns:
115 bool: RAM in GBs
116 """
117 global RAM_MEMO
118 if not RAM_MEMO:
119 out = subprocess.check_output("wmic OS get FreePhysicalMemory /Value", stderr=subprocess.STDOUT, shell=True)
120 RAM_MEMO = round(
121 int(str(out).strip("b").strip("'").replace("\\r", "").replace("\\n", "").replace("FreePhysicalMemory=", "")) / 1048576, 2
122 )

130def hashFile(filePath: str) -> str:
131 """Hashes a file
132
133 Args:
134 filePath (str): Path to the file to hash
135
136 Returns:
137 str: HEX string of the file's hash
138 """
139 if os.path.exists(filePath):
140 if available_ram() <= LOW_RAM:
141 sha256 = hashlib.sha256()
142 with open(filePath, "rb") as f:
143 while True:
144 data = f.read(BUF_SIZE)
145 if not data:
146 break
147 sha256.update(data)
148 return sha256.digest()
149 else:
150 with open(filePath, "rb") as f:
151 return hashlib.sha256(f.read()).hexdigest()
152 return ""
